private void EdWBFrm(string[] args)
        {
            /*
             * Edits the base form of the focused word.
             * Needs a word in focus.
             * Takes one argument: The new form
             */
            if (focus != null && focus.GetType() == typeof(Word))
            {
                if (args.Length == 1)
                {
                    string oldForm = ((Word)focus).GetForm();
                    central.EditWordForm((Word)focus, args[0]);
                    Console.WriteLine(string.Format("Changed the from from {0} to {1}.", oldForm, args[0]));
                }
                else
                {
                    Console.WriteLine("Incorrect number of arguments: This command takes 1 argument.\nType 'help edwbfrm' for more info.");
                }
            }
            else
            {
                Console.WriteLine("Needs a word in focus.\nType 'help focus' for more info.");
            }
        }
